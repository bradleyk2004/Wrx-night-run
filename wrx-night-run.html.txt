<canvas id="game" width="400" height="600"></canvas>

<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

let player, traffic, cops, score, speed, boost, boosting, gameOver;

function resetGame() {
  player = { x: 180, y: 480, w: 40, h: 80 };
  traffic = [];
  cops = [];
  score = 0;
  speed = 4;
  boost = 100;
  boosting = false;
  gameOver = false;
}

resetGame();

canvas.addEventListener("touchstart", e => {
  e.preventDefault();
  const t = e.touches[0];
  const rect = canvas.getBoundingClientRect();
  const x = t.clientX - rect.left;
  const y = t.clientY - rect.top;

  if (y < 150 && boost > 0) boosting = true;
  else if (x < canvas.width / 2 && player.x > 0) player.x -= 30;
  else if (x > canvas.width / 2 && player.x < 360) player.x += 30;
});

canvas.addEventListener("touchend", () => boosting = false);

function spawnTraffic() {
  traffic.push({ x: Math.random() * 360, y: -80, w: 40, h: 80 });
}

function spawnCop() {
  cops.push({ x: Math.random() * 360, y: -80, w: 40, h: 80 });
}

function hit(a, b) {
  return a.x < b.x + b.w &&
         a.x + a.w > b.x &&
         a.y < b.y + b.h &&
         a.y + a.h > b.y;
}

function update() {
  if (gameOver) return;

  ctx.clearRect(0,0,400,600);

  let runSpeed = boosting && boost > 0 ? speed + 4 : speed;
  if (boosting && boost > 0) boost--;
  else if (boost < 100) boost += 0.5;

  ctx.fillStyle = "#00bfff";
  ctx.fillRect(player.x, player.y, player.w, player.h);

  traffic.forEach(t => {
    t.y += runSpeed;
    ctx.fillStyle = "red";
    ctx.fillRect(t.x, t.y, t.w, t.h);
    if (hit(player, t)) gameOver = true;
  });

  cops.forEach(c => {
    c.y += runSpeed + 1;
    ctx.fillStyle = "blue";
    ctx.fillRect(c.x, c.y, c.w, c.h);
    if (hit(player, c)) gameOver = true;
  });

  traffic = traffic.filter(t => t.y < 600);
  cops = cops.filter(c => c.y < 600);

  score++;
  if (score % 60 === 0) spawnTraffic();
  if (score % 300 === 0) spawnCop();
  if (score % 200 === 0) speed += 0.5;

  ctx.fillStyle = "white";
  ctx.font = "16px Arial";
  ctx.fillText("Score: " + score, 10, 20);

  ctx.fillStyle = "yellow";
  ctx.fillRect(10, 30, boost * 1.5, 8);
}

function gameOverScreen() {
  ctx.fillStyle = "white";
  ctx.font = "28px Arial";
  ctx.fillText("BUSTED", 130, 260);
  ctx.font = "16px Arial";
  ctx.fillText("Refresh App to Restart", 110, 300);
}

function loop() {
  update();
  if (gameOver) gameOverScreen();
  requestAnimationFrame(loop);
}

loop();
</script>
